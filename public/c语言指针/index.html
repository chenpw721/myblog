#include<stdio.h>
int main(void){
    int a = 5;
    int *p;
    p = &a; //指针变量p指向普通变量a的地址
    print("*p = ", *p); //解引用
    print("p = ", p);
    print("&p = ", &p);
    return 0;
}

void类型的指针可以存储其他类型变量的地址，但不能解引用获取值
