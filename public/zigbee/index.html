/*初始化程序*/
void Initial(void){
    P1DIR |= 0x03;//P10、P11 定义为输出
    RLED = 1;
    YLED = 1;//LED
}

/*延时函数*/
void Delay(uint n){
    uint tt;
    for(tt = 0; tt < n; tt++);
    for(tt = 0; tt < n; tt++);
    for(tt = 0; tt < n; tt++);
    for(tt = 0; tt < n; tt++);
    for(tt = 0; tt < n; tt++);
}
